--Вывести список  логинов курьеров с количеством их заказов в статусе «В доставке» (поле inDelivery = true). 
SELECT c.login, count(o."courierId") FROM "Couriers" as c JOIN "Orders" as o On c.id = o."courierId"Where o."inDelivery" = true GROUP BY c.login;
--Вывести все трекеры заказов и их статусы.  
SELECT track, CASE WHEN finished THEN 2 WHEN cancelled THEN -1 WHEN "inDelivery" THEN 1 ELSE 0 END AS status FROM "Orders";
